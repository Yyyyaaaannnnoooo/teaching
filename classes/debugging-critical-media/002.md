function setup() {
	// Create a canvas using WEBGL renderer - this enables 3D graphics capabilities
	// The canvas fills the entire browser window
	createCanvas(windowWidth, windowHeight, WEBGL)
	
	// Set angle mode to DEGREES instead of radians
	// This makes rotation functions use 0-360 degrees rather than 0-2*PI
	angleMode(DEGREES)
	
	// Uncomment to set a fixed frame rate (currently disabled)
	//frameRate(1)

function draw() {
	// Set background color to dark gray (value 50 out of 255)
	background(50)
	
	// Enable mouse/touch orbit control
  // allows user to rotate the 3D scene
	// Parameter 1 sets the sensitivity of the rotation
	orbitControl(1)

	// Disable drawing outlines around shapes
	noStroke()
	
	// imageLight() creates a light source 
  // that projects the loaded image
	// This creates colored lighting based on the image's pixels
	imageLight(img);
	
	// Set the specular material color to magenta (255, 0, 255)
	// This determines how light reflects off shiny surfaces
	specularMaterial(255, 0, 255);
	
	// Set the shininess level 
  // (controls how concentrated the specular highlights are)
	// Higher values create smaller, more intense highlights
	shininess(100);
	
	// Uncomment to set ambient 
  // material color (currently disabled)
	// ambientMaterial(0, 255, 0);

	// Set fill color to white for the first box
	fill(255)
	
	// push() saves the current transformation 
  // state (position, rotation, scale)
	push()
	// Translate (move) to position (0, 0)
  // no movement from origin
	translate(0, 0)
	// Uncomment to rotate around X based on frame count
	rotateX(frameCount / 4)
	// Uncomment to rotate around Z based on frame count
	rotateZ(frameCount / 2)
	// Draw a 3D box with all sides equal to 50 units
	box(50)
	// pop() restores the previous transformation state
	pop()

	// Set fill color to blue for the second box
	fill(0, 0, 255)
	push()
	// Move to position (100, 100) on X and Y axes
  // Z remains at 0
	translate(100, 100)
	// Rotate around X axis based on frame count
	rotateX(frameCount / 4)
	// Rotate around Z axis based on frame count
	rotateZ(frameCount / 2)
	// Draw a 3D box with all sides equal to 50 units
	box(50)
	pop()
	
	// Call custom function to draw a box 
  // at (-100, -100) with size 200
	// Rotation amount increases with frameCount
  // red color (255,0,0)
	drawBox(-100, -100, 200, 70, frameCount / 10, 255, 0, 0)

	// Call custom function to draw a torus (donut shape)
	// Parameters: x, y, z position (all 0)
	// radius: 70, tube radius: 20
	// rotation based on frameCount, magenta color (255,0,255)
	drawTorus(
		0, 0, 0, // here I define position
		70, 20, // here I define sizes
		frameCount / 3, // here I define rotation
		255, 0, 255 // here I define Color
	)

	// Call custom function to draw a sphere
	// Position at origin (0,0,0)
	// Radius of 1000 (very large - will appear massive)
	// Rotation based on frameCount, green color (0,255,0)
	drawSphere(
		0, 0, 0, //position
		1000, // size
		frameCount / 10, // rotation
		0, 255, 0 // color
	)

}

// Custom function to draw a sphere with specified parameters
// x, y, z: position coordinates
// radius: size of the sphere
// rotY: rotation amount around Y axis
// r, g, b: RGB color values
function drawSphere(x, y, z, radius, rotY, r, g, b) {
	fill(r, g, b)     // Set the fill color
	push()            // Save current transformation state
	translate(x, y, z) // Move to specified position
	rotateY(rotY)      // Rotate around Y axis
	sphere(radius)     // Draw the sphere
	pop()              // Restore previous transformation state
}

// Custom function to draw a torus (donut shape)
// x, y, z: position coordinates
// radius: distance from center of torus to center of tube
// tubeRadius: radius of the tube itself
// rotZ: rotation amount around Z axis
// r, g, b: RGB color values
function drawTorus(x, y, z, radius, tubeRadius, rotZ, r, g, b) {
	fill(r, g, b)      // Set the fill color
	push()             // Save current transformation state
	translate(x, y, z)  // Move to specified position
	rotateZ(rotZ)       // Rotate around Z axis
	torus(radius, tubeRadius) // Draw the torus
	pop()               // Restore previous transformation state
}

// this function draws a box
// at position x, y, z
// with size: size
// rotX: rotation amount around X axis
// r, g, b: RGB color values
function drawBox(x, y, z, size, rotX, r, g, b) {
	fill(r, g, b)      // Set the fill color
	push()             // Save current transformation state
	translate(x, y, z)  // Move to specified position
	rotateX(rotX)       // Rotate around X axis
	box(size)           // Draw the box (cube) with all sides equal to 'size'
	pop()               // Restore previous transformation state
}
