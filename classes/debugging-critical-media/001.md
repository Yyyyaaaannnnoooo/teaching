## Good to know

p5.js runs in the browser by creating a `<canvas>` [element](https://developer.mozilla.org/en-US/docs/Web/HTML/Reference/Elements/canvas). The latter is an html element that allows for scriptable drawing on a web page.

P5.js takes advantage of this feature to draw programmatically on the web. The programming language is developed around functions that are named exactly for what they do: `square(100, 100, 50)` draws a square in position 100 on x-axis, 100 y-axis with a size of 50 pixels. All of the functions are referenced [here](https://p5js.org/reference/) and come with examples.

```javascript
// SETUP FUNCTION - runs once at the beginning
function setup() {
  // Creates a canvas that fills the entire browser window
  createCanvas(windowWidth, windowHeight)
  // Sets initial background to black
  background(0)

  // Sets rectangle drawing mode to CORNER (default)
  // This means x,y coordinates 
  // specify the top-left corner of rectangles
  rectMode(CORNER)
  
  // rectMode(CENTER) 
  // Would center the square on coordinates
}

// Global variables for position and size of the main square
let x = 10  // x-coordinate position
let y = 10  // y-coordinate position
let size = 100  // size of the square (width and height)



function draw() {
  // DRAW FUNCTION - runs continuously (about 60 times per second)
  
  // Clears the canvas with black each frame (creates animation effect)
  background(0)

  noFill()
  // rotate a square (commented out code block)
  // This section would create a rotating square if uncommented
  stroke(255)  // Sets stroke color to white

  // move a square across the canvas
  x = x + 2  // Increments x position by 2 pixels each frame (moves right)
  y = 10  // Fixed y position at 10 pixels from top
  square(x, y, size)  // Draws the square at current (x,y) position
  
  // return the square to the beginning
  if(x > width) {  // If square moves beyond right edge of canvas
    x = 0  // Reset to left edge
  }

  // multiple square marching on the screen
  // Creates a grid of squares moving horizontally
  // Outer loop: creates 5 rows
  for(let i = 0; i < 5; i++) {
    // Calculates vertical position for each row
    let h = y * (i + 1) * 13  
    // h increases with each row: row1: 130, row2: 260, etc.
    // it is also possible to nest loops
    for(let j = 0; j < 3; j++) {  
      // Inner loop: creates 3 squares per row
      w = x + (j+1)* (size + 20)  
      // Calculates horizontal position
      // w creates spacing between squares in each row
      // Squares are placed with 120px gaps (size 100 + 20)
      
      square(w, h, size)  
      // Draws square at calculated position
    }
  }
  // Total squares drawn: 5 rows × 3 columns = 15 squares
}
```

## Adding Javascritpt to neocities

### On the main page

to add javascript to your main neocities page, you need to add a `div` that will be the container for the p5.js sketch anywhere within the `body` tag.

```html
<!-- 
  p5.js canvas container div
  - This div acts as a container for the p5.js canvas element
  - The p5.js library will attach the canvas to this div
  - when declared in the function setup()
  //////////// JS ////////////
  - let canvas = createCanvas(600, 400);
  - canavs.parent("#p5");
  ///////////// END ///////////
  - Without this div, the canvas would be appended to the end of the body
  - The fixed positioning allows the canvas to float over other content
-->
<div id="p5"></div>
```

Then you need to add some styling to the #p5 selector like this:

```css
/* CSS styling for the p5.js canvas container */
#p5 {
  z-index: 99999999999; 
  /* Extremely high stacking order - ensures canvas appears above almost all other elements */
  position: fixed; 
  /* Fixes the element relative to the viewport (stays in place when scrolling) */
  top: 200px; 
  /* Positions the element 200px from the top of the viewport */
  left: 200px; 
  /* Positions the element 200px from the left of the viewport */
}
```

    /*	_webgl_primatives // cc teddavis.org 2019	*/

    let v = 0.3;

    function setup() {
      createCanvas(windowWidth, windowHeight, WEBGL);
      angleMode(DEGREES);
    }

    function draw() {
      background(0);
      orbitControl(5);

      ortho();

      if (mouseIsPressed) {
        lights();
        noStroke();
        fill(255);
      } else {
        stroke(255);
        noFill();
      }

      push();
      translate(rPos(0));
      rotateX(frameCount / 4);
      rotateZ(frameCount / 2);
      plane(150, 150);
      pop();

      push();
      translate(rPos(50));
      rotateX(frameCount / 4);
      rotateZ(frameCount / 2);
      box(150);
      pop();

      push();
      translate(rPos(100));
      rotateX(frameCount / 4);
      rotateY(frameCount / 4);
      sphere(100);
      pop();

      push();
      translate(rPos(150));
      rotateX(frameCount / 4);
      rotateY(frameCount / 4);
      ellipsoid(100, 30);
      pop();

      push();
      translate(rPos(200));
      rotateY(frameCount / 2);
      rotateX(frameCount / 3);
      cone(100, 150);
      pop();

      push();
      translate(rPos(250));
      rotateX(frameCount / 4);
      rotateZ(frameCount / 3);
      torus(100, 50);
      pop();

      push();
      translate(rPos(300));
      rotateY(frameCount / 2);
      rotateZ(frameCount / 4);
      cylinder(100, 100);
      pop();
    }

    function rPos(off) {
      let x = (sin(off + frameCount * v) * width) / 3;
      let y = (cos(off + frameCount * v) * height) / 3;
      return createVector(x, y);
    }
